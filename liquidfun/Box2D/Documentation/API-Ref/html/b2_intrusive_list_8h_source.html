67  void InsertAfter(b2IntrusiveListNode *node) {
68  b2Assert(!node->InList());
69  node->m_next = m_next;
70  node->m_prev = this;
71  m_next->m_prev = node;
72  m_next = node;
73  }
74 
75  void InsertBefore(b2IntrusiveListNode *node) {
76  b2Assert(!node->InList());
77  node->m_next = this;
78  node->m_prev = m_prev;
79  m_prev->m_next = node;
80  m_prev = node;
81  }
82 
83  void Remove() {
84  m_prev->m_next = m_next;
85  m_next->m_prev = m_prev;
86  Initialize();
87  }
88 
89  bool InList() const {
90  return m_next != m_prev;
91  }
92 
93  int32 GetLength() const {
94  int32 len = 0;
95  for (const b2IntrusiveListNode* n = GetNext();
96  n != GetTerminator(); n = n->GetNext()) {
97  len++;
98  }
99  return len;
100  }

103  b2IntrusiveListNode* GetPrevious() { return m_prev; }
104  b2IntrusiveListNode* GetTerminator() { return this; }
105  const b2IntrusiveListNode* GetNext() const { return m_next; }
106  const b2IntrusiveListNode* GetPrevious() const { return m_prev; }
107  const b2IntrusiveListNode* GetTerminator() const { return this; }
